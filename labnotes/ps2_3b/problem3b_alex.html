clear; close all; clc;

Generate white noise stimulus:

p = 25;      % spatial dimensions
T = 5e6;     % number of time points
x = linspace(-0.5,0.5,p); % spatial axis
s = randn(p,T);

Define Edge Filter

a = 20;  b = 0.02;
f = a*(x.^3) .* exp(-x.^2/b);

Compute the firing rate with a quadratic non-linearity introduced

theta = 0.01;  D = 0.01;  r0 = 1;
r = r0.*(1+exp( (theta-(f*s).^2)./D) ).^-1; % instantaneous firing rate
spk = poissrnd(r);                          % generate Poisson spikes
figure;
subplot(2,1,1)
plot(r(1:100))
title('firing rate (first 100 time points)')
subplot(2,1,2)
imagesc(s(:,1:100))
colormap bone
title('stimulus (first 100 time points)')

Attempt to compute the STA (it doesn't work)

Stimuli that invoked multiple spikes are counted multiple times by repeating them in the matrix s_spk. This makes the STA a weighted average.

pspk = sum(spk)./length(spk);  % probability of having a spike in any time window
s_spk = [s(:,spk>0), s(:,spk>1), s(:,spk>2), s(:,spk>3), s(:,spk>4),...
         s(:,spk>5), s(:,spk>6), s(:,spk>7), s(:,spk>8), s(:,spk>9)];
sta = mean(s_spk,2);           % average all stimuli that coincided with spikes

figure; hold on
plot(x,f ./ norm(f),'-b')
plot(x,sta ./ norm(sta),'-r')
xlabel('spatial dimension')
ylabel('stimulus intensity')
legend('Edge Filter (f)','Spike-Triggered Average','location','northwest')

figure; hold on
plot(f*s(:,1:5e3),r(1:5e3),'xb')
plot([f*sta; f*sta],get(gca,'ylim')','--g')
xlim([-0.2 0.2])
ylabel('firing rate')
xlabel('magnitude of filtered stimulus')
legend('Each Stimulus','STA','location','north')

Compute Spike-Triggered Covariance (works!)

nspk = sum(spk);
sta_matrix = repmat(sta,1,size(s_spk,2));
Cstc = (1/(nspk-1)) .* (s_spk-sta_matrix) * (s_spk-sta_matrix)';
stim_mean = repmat(mean(s,2),1,T);
Cprior = (1/(T-1)) .* (s-stim_mean) * (s-stim_mean)';
deltaC = Cstc-Cprior;
[E,L] = eig(deltaC); % E = eigenvectors; L = diagonal eigenvalue matrix
[max_eigVal,max_ind] = max(abs(diag(L)));
stc1 = E(:,max_ind)'; % Use eigenvector with the largest change in covariance

figure;
imagesc(Cprior)
title({'Stimulus Covariance Matrix',...
       'the stimulus is white noise, so there is no covariance'});

figure;
imagesc(Cstc)
title({'Spike Triggered Covariance Matrix',...
       'there are small patterns visible, but they are overpowered by the white noise'});

figure;
imagesc(deltaC)
title({'Spike Triggered Covariance Matrix, after subtracting stimulus covariance matrix.',...
       'The correlation pattern is indicative of the edge filter used to generate our data'});

figure; hold on
plot(x,f ./ norm(f),'-b')
plot(x,stc1 ./ norm(stc1),'-r')
title({'Note: STC produces a direction of maximal variance.',...
      'On some trials the red line will be a reflection of the blue'})
xlabel('spatial dimension')
ylabel('stimulus intensity (normalized)')
legend('Edge Filter (f)','Spike-Triggered Covariance Estimate','location','northwest')

Estimate the firing rate using Bayes theorem and the STC

Variable definitions: "spk_prior" is , "stim_prior" is , "likli" is 

% spk_prior is the same as the overall mean firing rate.
spk_prior = mean(r);

% stim_prior is a histogram of the projections of s(:,t) onto the direction
% identified by the spike-triggered covariance matrix. Note that since the
% sta is equal to zero, we ignore it in our model.
proj = stc1*s;
proj_bin_size = range(proj)./1e2;
stim_prior = @(x) sum(proj<(x+proj_bin_size) & proj>(x-proj_bin_size))./T;

% The liklihood function is similar to the stimulus prior, but selects out
% stimuli that actually elicited one or more spikes.
proj_spk = stc1*s_spk;
likli = @(x) sum(proj_spk<(x+proj_bin_size) & proj_spk>(x-proj_bin_size))./size(s_spk,2);

% Non-linear function r_est(s*f_est) relates the projected stimulus into an
% estimated firing rate by Bayes Theorem:
r_est = @(x) likli(x).*spk_prior./stim_prior(x);

for aa = 1:100
  r_est_100(aa) = r_est(stc1*s(:,aa)); %#ok<SAGROW>
end

% Compare predicted to actual firing rate
figure;
subplot(2,1,1); hold on
plot(r(1:100),'-b')
plot(r_est_100,'-r')
legend('actual firing rate','p(spike|s*f_{est})')
title({'firing rate (first 100 time points)','The fit is nearly perfect - blue line is hidden beneath the red'})
subplot(2,1,2)
imagesc(s(:,1:100))
colormap bone
title('stimulus (first 100 time points)')

Compare predicted and actual non-linearity.

The fit is not excellent. Which is interesting, given that our reconstruction of r(t) in the previous figure is quite good.

figure; hold on
proj_levels = linspace(min(min(f*s)),max(max(f*s)),100);
for aa = 1:length(proj_levels)
  r_est_100(aa) = r_est(proj_levels(aa)); %#ok<SAGROW>
end
plot(f*s(:,1:5e3),(r(1:5e3)-min(r(1:5e3)))./range(r(1:5e3)),'xb')
plot(proj_levels,(r_est_100-min(r_est_100))./range(r_est_100),'-r')
xlim([-0.2 0.2])
xlabel('Projection onto stimulus filter')
ylabel('firing rate')
legend('actual','estimated','location','North')
